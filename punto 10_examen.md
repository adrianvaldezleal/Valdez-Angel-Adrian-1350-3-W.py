#Leer un valor N
N = int(input("Introduce un n√∫mero entero para calcular su factorial: "))

#Calcular el factorial
factorial = 1
for i in range(1, N + 1):
    factorial *= i

#Mostrar el resultado
print(f"El factorial es", factorial)
